def tail(n = nil, &block)
  super(n) do |ln|
    $stderr.puts ln
    block.call(ln) if block
  end
rescue Interrupt => e
  nil
end

def tail_to_eof

  tail
ensure
  return_if_eof = orig_return_if_eof
end

def tail_until(rx)
  self.tail do |ln|
    raise IO::Tail::ReturnException if ln =~ rx
  end
end
